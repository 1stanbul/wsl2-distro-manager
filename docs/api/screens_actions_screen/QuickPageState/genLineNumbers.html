void genLineNumbers(int offset) {
  for (int i = 1 + offset; i < lineNum + offset + 1; i++) {
    lineNumbers += i.toString();
    if (i < lineNum + offset) {
      lineNumbers += '\n';
    }
  }
  setState(() {
    lineNumbers = lineNumbers;
  });
}
